SELECT
    transactions.id AS transation_id,
    bouteille_id,
    cellier_id,
    quantite,
    date_achat,
    garde_jusqua,
    notes,
    prix,
    millesime,
    bouteilles.id AS bouteille_id,
    bouteilles.nom,
    bouteilles.code_saq,
    bouteilles.pays,
    bouteilles.description,
    bouteilles.prix_saq,
    bouteilles.url_image,
    bouteilles.type_id,
    TYPES.id AS type_id,
    TYPE
FROM
    `transactions`
JOIN bouteilles ON bouteille_id = transactions.bouteille_id = bouteilles.id
JOIN TYPES ON type_id = bouteilles.id = TYPES.id
